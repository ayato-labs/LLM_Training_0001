# 詳細なフローチャート図

本システムにおけるデータの抽出から、作品単位の分割、**「下限 30M Proxy モデルから目標 150M モデルへのスケール則応用ロジック」**、Chinchilla/μP則ベースの多次元HPO、パイロット自動検証、DeepSpeed学習、多角評価までの全ライフサイクルを表したフローチャートです。

```mermaid
graph TD
    %% スタイル定義
    classDef data fill:#f9f,stroke:#333,stroke-width:2px;
    classDef process fill:#bbf,stroke:#333,stroke-width:1px;
    classDef active fill:#fbb,stroke:#333,stroke-width:2px;
    classDef daemon fill:#bfb,stroke:#333,stroke-width:1px;
    classDef scale fill:#ffeb3b,stroke:#333,stroke-width:2px;
    classDef pilot fill:#f0f,stroke:#333,stroke-width:2px;

    %% --------------------------------------------------------
    %% 1. データ抽出・クレンジングフェーズ
    %% --------------------------------------------------------
    subgraph Data_Extraction ["1. Raw Data Extraction & Cleaning"]
        DB["SQLite: novels.db"]:::data -->|"RAWテキスト取得"| ED["extract_data.py"]:::process
        ED -->|"ルビ除去 《.*?》"| ED_Clean["テキストクレンジング"]:::process
        ED_Clean -->|"50文字未満の短編を除外"| ED_Filter["データフィルタリング"]:::process
        ED_Filter -->|"保存"| CORPUS["data/corpus.jsonl"]:::data
    end

    %% --------------------------------------------------------
    %% 2. 前処理・トークナイズフェーズ
    %% --------------------------------------------------------
    subgraph Data_Preparation ["2. Dataset Splitting & Preprocessing"]
        CORPUS -->|"タイトルをキーにグルーピング"| SD["split_dataset.py"]:::process
        SD -->|"決定論的シャッフル/データリーク防止"| SD_Split["小説単位での分離 99% : 1%"]:::process
        SD_Split -->|"出力"| T_DATA["data/train_dataset.jsonl"]:::data
        SD_Split -->|"出力"| V_DATA["data/val_dataset.jsonl"]:::data

        CORPUS -->|"テキスト抽出"| TT["train_tokenizer.py"]:::process
        TT -->|"SentencePiece BPE 訓練"| SP_Train["Vocab 64,000 / 特殊トークン定義"]:::process
        SP_Train -->|"HuggingFace形式へ変換"| TOK_JSON["data/tokenizer.json"]:::data
    end

    %% --------------------------------------------------------
    %% 3. 【核心部】スケール則応用 ＆ 多次元Proxyパラメータ探索フェーズ
    %% --------------------------------------------------------
    subgraph Scaling_And_Proxy_HPO ["3. Scaling Law & Multi-Dimensional Proxy HPO"]
        MAIN["src/main.py"]:::active -->|"ベース設定読込 (YAML駆動)"| CFG["config.yaml / training_config.py"]:::data
        
        %% ユーザー固有のスケーリングパラメータ定義
        CFG -->|"目標: TARGET_PARAMS (config.yaml)"| TARGET["目標モデル定義"]:::scale
        CFG -->|"Proxy: max(TARGET*0.05, 30M)"| PROXY["Proxyモデル動的生成"]:::scale
        
        %% Proxy層での多次元探索
        TARGET -->|"N_target / D_tokens"| SCALE_LAW["Scaling Law Engine"]:::scale
        PROXY -->|"N_proxy / L=12 H=12固定"| PROXY_MODEL["Proxyモデル動的生成"]:::scale
        
        %% Scaling Law Engine
        T_DATA -->|"行数から正確な総トークン数Dを算出"| COMP_FP["compute_dataset_fingerprint"]:::process
        COMP_FP -->|"総トークン数D, N_target, N_proxy"| SCALE_LAW["Scaling Law Engine"]:::scale
        
        %% Scaling Law Engine 内部
        subgraph Scaling_Law_Engine ["Scaling Law Engine (Chinchilla/μP/μTransfer)"]
            SL_CHIN["Chinchilla則: η* ∝ N^-0.35, B* ∝ N^0.5"]:::scale
            SL_MU_P["μP理論: LR 2D/1D分離, grad_clip=1.0"]:::scale
            SL_MUT["μTransfer: Proxy→本番へハイパラ転移"]:::scale
        end
        
        PROXY_MODEL -->|"動的構成: L=12, H=12, H_dim=自動"| HPO_SEARCH["多次元HPO探索 (Optuna)"]:::active
        
        %% 多次元HPO探索
        subgraph Multi_Dim_HPO ["多次元HPO探索 (Optuna + Scaling Law Priors)"]
            HPO_INPUT["Scaling Law Priors\nLR: N^-0.35, BS: N^0.5\nWD: N^-0.1, Beta2, Warmup"]:::scale
            HPO_SAMPLER["TPE(multivariate=True)"]:::process
            HPO_PRUNER["MedianPruner (早期打ち切り)"]:::process
            HPO_SEARCH -->|"7次元探索: LR_2D, LR_1D, WD, β2, Warmup, GradClip, BS"| HPO_SEARCH
            HPO_SEARCH -->|"Proxy Model (L=12, H=12, H_dim動的)"| TRAIN_PROXY["Proxy Training (30M)"]:::process
            TRAIN_PROXY -->|"Loss反転"| OPTUNA_PRUNER["MedianPruner: 早期打ち切り"]:::process
        end
        
        %% 統合
        PROXY_MODEL & SCALE_LAW -->|"Optuna Study (multivariate TPE + MedianPruner)"| HPO_CFG["最適ハイパラ結合Config"]:::data
    end

    %% --------------------------------------------------------
    %% 4. 【必須】パイロット自動検証フェーズ
    %% --------------------------------------------------------
    subgraph Pilot_Check ["4. Pilot Auto-Check (本番前検証)"]
        HPO_CFG -->|"Best Params + Step Law"| PILOT["Pilot Auto-Check"]:::pilot
        PILOT -->|"本番モデル(150M) で 50steps / 0.5%データ"| PILOT_RUN["Pilot Training (50 steps / 0.5% data)"]:::pilot
        PILOT_RUN -->|"Loss NaN/Inf → FAIL"| PILOT_FAIL{"Judgment"}:::pilot
        PILOT_RUN -->|"Loss > 50 → WARNING"| PILOT_WARN
        PILOT_RUN -->|"正常 → PASS"| PILOT_PASS
        PILOT_FAIL -->|"本番中止"| STOP["Exit(1)"]:::active
        PILOT_WARN -->|"警告ログ出力後継続"| CONT
        PILOT_PASS -->|"GO"| CONT
        CONT -->|"GO"| TRAIN_PHASE["5. Full Training Loop"]:::active
    end

    %% --------------------------------------------------------
    %% 5. 学習実行ループフェーズ
    %% --------------------------------------------------------
    subgraph Training_Loop ["5. Full Training Execution Loop"]
        HPO_CFG -->|"最適ハイパラ + Pilot PASS"| T_MOD["train_model.py: train"]:::active
        
        T_MOD -->|"環境検知"| SNAP["env_snapshot.py: capture_env_snapshot"]:::process
        T_MOD -->|"固定・決定論的アルゴリズム有効化"| SEED["set_seed.py: set_seed"]:::process
        
        SNAP -->|"Gitハッシュ/主要パッケージバージョン"| ML_INIT["MLflow: start_run"]:::process
        SEED -->|"シード固定情報注入"| ML_INIT
        
        T_MOD -->|"データ読込"| LD["load_dataset"]:::process
        T_DATA & V_DATA --> LD
        TOK_JSON -->|PreTrainedTokenizerFast| LD
        
        LD -->|"padding='max_length' / Packed Sequence化"| TM["Dataset.map: tokenize_function"]:::process
        TM -->|"モデルアーキテクチャ定義"| L_CFG["LlamaConfig"]:::process
        L_CFG -->|"インスタンス化"| L_MOD["LlamaForCausalLM"]:::process
        
        T_MOD -->|"VRAM容量に応じた最適化割り当て"| DS_GEN["generate_deepspeed_config"]:::process
        DS_GEN -->|"BF16 / ZeRO Stage 1-3 / CPU Offloadの判定"| DS_CFG["temp_ds_config.json"]:::data
        
        L_MOD & DS_CFG -->|"Trainer構築"| CT["CustomTrainer"]:::process
        
        subgraph Custom_Optimizer ["Muon/AdamW スプリット最適化 (HPO結果反映)"]
            CT -->|"len.shape >= 2 の行列パラメータ (2D)"| MUON["Muon / NorMuon (lr_2d, momentum)"]:::process
            CT -->|"Embedding / Bias / LayerNorm (1D)"| ADAMW["AdamW (lr_1d, weight_decay, beta2)"]:::process
        end
        
        subgraph WSD_LR_Schedule ["WSD Scheduler 制御ループ"]
            CT -->|"全ステップの 1%"| W_PHASE["Warmup: Linear 増加"]:::process
            CT -->|"全ステップの 80%"| S_PHASE["Stable: Peak * 0.55 維持"]:::process
            CT -->|"全ステップの 20%"| D_PHASE["Decay: 1 - 階乗根 減衰"]:::process
        end
        
        subgraph Pilot_Mode_Switch ["Pilot / 本番 切替"]
            CT -->|"pilot_mode=True"| PILOT_ARGS["steps=50, data=0.5%, no ckpt/eval"]:::pilot
            CT -->|"pilot_mode=False"| FULL_ARGS["full steps, full data, ckpt/eval/enabled"]:::active
        end
        
        CT -->|"Micro Batch + Gradient Accumulation"| DS_TRAIN["DeepSpeed Training Loop"]:::active
        DS_TRAIN -->|"5ステップ毎(Pilot) / 10ステップ毎(本番)"| LOG_STEP["TensorBoard / MLflow へ指標記録"]:::process
        
        DS_TRAIN -->|"1000ステップ毎"| SAVE_CKPT["ローカルへのチェックポイント保存"]:::process
        SAVE_CKPT -->|"イベント発火"| DRIVE_CB["DriveUploadCallback: on_save"]:::process
    end

    %% --------------------------------------------------------
    %% 5. 評価・インファレンスフェーズ
    %% --------------------------------------------------------
    subgraph Evaluation_Suite ["6. Evaluation & Benchmarking"]
        DS_TRAIN -->|"学習終了後"| FIN_SAVE["最終モデル & トークナイザーの永続化"]:::process
        FIN_SAVE -->|"モデルロード"| EVAL_M["eval_inference/evaluate_model.py"]:::process
        FIN_SAVE -->|"モデルロード"| BENCH["evaluation/benchmark.py"]:::process
        
        EVAL_M -->|"特殊トークンメタデータ付きプロンプト入力"| INFERENCE["テキスト生成制御テスト TC-01~05"]:::process
        INFERENCE -->|"SentencePiece固有スペースのクレンジング"| CLEAN_OUT["構造化評価レポート出力"]:::process
        CLEAN_OUT -->|"保存"| EVAL_REP["logs/eval_report_*.md / eval_results_*.json"]:::data
        
        BENCH -->|"スタイル・感情・キャラクター一貫性・文法"| MEASURE["指標の定量計測"]:::process
        MEASURE -->|"保存"| BENCH_RES["logs/benchmark_*.json"]:::data
        
        EVAL_REP & BENCH_RES -->|"実験結果を完全統合"| ML_CLOSE["MLflow: log_metrics / end_run"]:::process
        
        ML_CLOSE -->|"過去ログ比較"| COMP_RUN["evaluation/compare_runs.py"]:::process
        COMP_RUN -->|"マルチシード評価値入力"| STATS["evaluation/statistics.py"]:::process
        STATS -->|"Welchのt検定 / Cohen's d効果サイズ計算"| STATS_RES["有意差検定付き比較レポート"]:::data
        
        ML_CLOSE -->|"Run ID指定"| REPROD["evaluation/reproduce.py"]:::process
        REPROD -->|"Gitチェックアウト＆DVCプル自動化"| SH_SCRIPT["完全再現用シェルスクリプト / .bat"]:::data
    end

    %% --------------------------------------------------------
    %% 6. 非同期クラウドバックアップ・ストレージ管理デーモン
    %% --------------------------------------------------------
    subgraph Storage_Management ["6. Cloud Archiving & Cleanup Daemon"]
        DRIVE_CB -->|"非同期処理キュー挿入"| DAEMON["utils/drive_uploader.py: Daemon"]:::daemon
        
        DAEMON -->|"30秒間隔のポーリング"| CHK_DIR["モデル出力ディレクトリ監視"]:::daemon
        CHK_DIR -->|"未アップロードのチェックポイント検出"| COMPRESS["shutil.make_archive: ZIP圧縮"]:::daemon
        COMPRESS -->|"Resumable Media Upload"| GD_UPLOAD["Google Drive: Novel_LLM_Checkpoints"]:::daemon
        GD_UPLOAD -->|"成功"| MARK_UP[".uploaded フラグファイルの生成"]:::daemon
        
        DAEMON -->|"定期実行: 60秒毎"| LOG_BAK["mlruns / TensorBoard ログの同期"]:::daemon
        DAEMON -->|"定期実行: 60秒毎"| DOC_BAK["docs/ADR ドキュメントの同期"]:::daemon
        DAEMON -->|"定期実行: 60秒毎"| REP_BAK["logs/評価レポートの同期"]:::daemon
        DAEMON -->|"定期実行: 60秒毎"| DVC_BAK[".dvc/cache 容量制限内バックアップ"]:::daemon
        DAEMON -->|"定期実行: 60秒毎"| MODEL_BAK["最終成果物モデルの専用フォルダ同期"]:::daemon
        
        MARK_UP -->|"定期クリーンアップ: 5分毎"| CLEAN_LOCAL["cleanup_old_checkpoints"]:::daemon
        CLEAN_LOCAL -->|"ローカルの保持上限 LOCAL_CHECKPOINT_KEEP を超えた古いフォルダを削除"| SSD_FREE["ローカルストレージ解放"]:::daemon
    end
